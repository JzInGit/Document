1. 恢复目录
由RMAN使用、维护的用来放置备份信息的仓库，RMAN利用恢复目录记载的信息去判断如何执行需要的备份恢复操作。

--  创建用于恢复的表空间
    CREATE TABLESPACE RMAN_TS DATAFILE 'D:\ORACLE\ORADATA\RMAN\RMAN_TS.DBF' SIZE 128M;
--  创建用于恢复的用户
    CREATE USER RMAN IDENTIFIED BY RMAN DEFAULT TABLESPACE RMAN_TS TEMPORARY TABLESPACE TEMP QUOTA UNLIMITED ON RMAN_YS;
--  赋予用户权限
    GRANT RECOVERY_CATALOG_OWNER,CONNECT,RESOURCE TO RMAN;
--  登录恢复目录
    RMAN CATALOG RMAN/RMAN
--  安装恢复目录
    CREATE CATALOG TABLESPACE RMAN_TS;
--  环境变量：
NLS_LANG=SIMPLIFIED CHINESE_CHINA.ZHS16GBK
NLS_DATE_FORMAT=DD-MON-YYYY HH24:MI:SS
